POST /user/login			Data: username=text&password=text
POST /user/logout

//FOR EVENT MANIPULATION
POST /event 				Data: {event attributes...}
PUT /event 					Data: {event attributes to update..., id: <event id>}
DELETE /event 				Data: {id: <event id>}
